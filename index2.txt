    <script>
        // Accordion toggle functionality
        function toggleAccordion(header) {
            const accordion = header.parentElement;
            accordion.classList.toggle('active');
        }

        // Decision button selection
        function selectDecision(button, value) {
            // Remove selected class from all decision buttons
            const buttons = document.querySelectorAll('.decision-btn');
            buttons.forEach(btn => btn.classList.remove('selected'));
            
            // Add selected class to clicked button
            button.classList.add('selected');
            
            // Update hidden input value
            document.getElementById('decisionInput').value = value;
        }

        // Submit form
       function handleSubmit(event) {
    event.preventDefault(); // Prevent default form submit

    const form = document.getElementById('leanForm');

    // Validate required fields
    if (!form.checkValidity()) {
        form.reportValidity();
        return;
    }

    // Check if decision is selected
    if (!document.getElementById('decisionInput').value) {
        alert('Please select a decision (Review, Reject, or Hold)');
        return;
    }

    // Collect form data
    const formData = new FormData(form);

    // Send to Formspree
    fetch(form.action, {
        method: 'POST',
        body: formData,
        headers: {
            'Accept': 'application/json'
        }
    }).then(response => {
        if (response.ok) {
            // Show success message
            document.getElementById('formContent').classList.add('hide');
            document.getElementById('successMessage').classList.add('show');

            // Reset form after 3 seconds
            setTimeout(function() {
                form.reset();
                document.querySelectorAll('.decision-btn').forEach(btn => btn.classList.remove('selected'));
                document.getElementById('decisionInput').value = '';
                document.getElementById('successMessage').classList.remove('show');
                document.getElementById('formContent').classList.remove('hide');
            }, 3000);
        } else {
            alert('There was a problem submitting your form. Please try again.');
        }
    }).catch(error => {
        console.error(error);
        alert('There was a problem submitting your form. Please try again.');
    });
}


        // Save form (stores data in localStorage)
        function saveForm() {
            const form = document.getElementById('leanForm');
            const formData = new FormData(form);
            const data = Object.fromEntries(formData.entries());
            
            // Add decision value
            data.decision = document.getElementById('decisionInput').value;
            
            // Save to localStorage
            localStorage.setItem('leanFormData', JSON.stringify(data));
            
            alert('Form data saved successfully!');
        }

        // Cancel form
        function cancelForm() {
            if (confirm('Are you sure you want to cancel? Any unsaved changes will be lost.')) {
                document.getElementById('leanForm').reset();
                document.querySelectorAll('.decision-btn').forEach(btn => btn.classList.remove('selected'));
                document.getElementById('decisionInput').value = '';
            }
        }

        // Load saved form data on page load
        window.addEventListener('load', function() {
            const savedData = localStorage.getItem('leanFormData');
            if (savedData) {
                const data = JSON.parse(savedData);
                const form = document.getElementById('leanForm');
                
                // Populate form fields
                Object.keys(data).forEach(key => {
                    const field = form.elements[key];
                    if (field) {
                        field.value = data[key];
                    }
                });
                
                // Set decision button
                if (data.decision) {
                    const decisionButtons = document.querySelectorAll('.decision-btn');
                    decisionButtons.forEach(btn => {
                        if (btn.textContent.toLowerCase() === data.decision) {
                            btn.classList.add('selected');
                        }
                    });
                    document.getElementById('decisionInput').value = data.decision;
                }
            }
        });
    </script>
